51 public class AddSubCommand extends AbstractCommand implements IExecutableCommand {
52 
53  @Localizable static final String _ONE_ONLY = "You must provide only one item stack.";
54  @Localizable static final String _ITEM_ALREADY_EXISTS = "There is already an item with the name '{0}'.";
55  @Localizable static final String _FAILED = "Failed to add item.";
56  @Localizable static final String _SUCCESS = "Quest item '{0}' created.";
57 
58  @Override
59  public void execute (CommandSender sender, ICommandArguments args) throws CommandException {
60 
61  String itemName = args.getName("itemName", 48);
62  ItemStack[] items = args.getItemStack(sender, "item");
63 
64  if (items.length != 1)
65  throw new CommandException(Lang.get(_ONE_ONLY));
66 
67  ScriptItemManager manager = ArborianQuests.getScriptItemManager();
68 
69  ScriptItem item = manager.get(itemName);
70  if (item != null)
71  throw new CommandException(Lang.get(_ITEM_ALREADY_EXISTS, item.getName()));
72 
73  if (manager.add(itemName, items[0]) == null)
74  throw new CommandException(Lang.get(_FAILED));
75 
76  tellSuccess(sender, Lang.get(_SUCCESS, itemName));
77  }
78 }
