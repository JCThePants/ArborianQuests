274  private class DialogRunner {
275 
276  boolean isCancelled;
277 
278  void run() {
279 
280  if (_isDisposed) {
281  cancel();
282  return;
283  }
284 
285  if (!_player.isOnline()) {
286  end();
287  return;
288  }
289 
290  if (_padTicksLeft > 0)
291  _padTicksLeft--;
292 
293  if (_ticksLeft > 0) {
294  _ticksLeft--;
295  return;
296  }
297 
298  if (_currentIndex >= _actions.size()) {
299 
300  if (_padTicksLeft <= 0)
301  cancel();
302 
303  return;
304  }
305 
306  IDialogAction action = _actions.get(_currentIndex);
307  _ticksLeft = action.duration();
308 
309  try {
310  action.run(_player, DialogContext.this);
311  }
312  catch (Throwable e) {
313  e.printStackTrace();
314  }
315 
316  _currentIndex++;
317  }
318 
319  void cancel() {
320 
321  this.isCancelled = true;
322  _runners.remove(this);
323 
324  if (_player != null) {
325  DialogSessions.remove(_player, DialogContext.this);
326  _agents.update("onComplete", DialogContext.this);
327  }
328 
329  _player = null;
330  _ticksLeft = 0;
331  _padTicksLeft = 0;
332  _currentIndex = 0;
333  }
334 
335  void start() {
336  this.isCancelled = false;
337  _runners.add(this);
338  _agents.update("onStart", this);
339  }
340  }
