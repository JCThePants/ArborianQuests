424  // Set the quest status of a player
425  private void setStatus(UUID playerId, QuestStatus status) {
426 
427  if (status == QuestStatus.NONE) {
428  _playerNodes.remove(playerId.toString());
429  }
430  else {
431  _playerNodes.set(playerId.toString() + ".status", status);
432  }
433 
434  if (status.getCurrentStatus() == CurrentQuestStatus.NONE) {
435  _playerQuests.remove(playerId, this);
436  }
437  else if (status.getCurrentStatus() == CurrentQuestStatus.IN_PROGRESS) {
438  _playerQuests.put(playerId, this);
439  }
440 
441  _playerNodes.save();
442  }

444  // initial settings load
445  private void loadSettings() {
446 
447  // load players
448  for (IDataNode playerNode : _playerNodes) {
449  UUID id = TextUtils.parseUUID(playerNode.getName());
450  if (id == null)
451  continue;
452 
453  QuestStatus status = playerNode.getEnum("", QuestStatus.NONE, QuestStatus.class);
454  //noinspection ConstantConditions
455  if (status.getCurrentStatus() != CurrentQuestStatus.IN_PROGRESS)
456  continue;
457 
458  _playerQuests.put(id, this);
459  }
460 
461  // load sub quests

463 
464  String questName = questNode.getName();
465
