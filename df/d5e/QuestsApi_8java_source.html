95  public static Quest getQuest(String questPath, boolean isNullAllowed) {
96  PreCon.notNullOrEmpty(questPath, "questPath");
97 
98  Quest quest = PATH_CACHE.get(questPath);
99  if (quest != null)
100  return quest;
101 
102  quest = Quest.getQuestFromPath(questPath);
103  if (quest == null && isNullAllowed)
104  return null;
105 
106  PreCon.isValid(quest != null, "Quest path '{0}' not found.", questPath);
107 
108  PATH_CACHE.put(questPath, quest);
109 
110  return quest;
111  }

401  public void queryQuest(Object player, String questPath, final Runnable onAccept) {
402  PreCon.notNull(player, "player");
403  PreCon.notNullOrEmpty(questPath, "questPath");
404  PreCon.notNull(onAccept, "onAccept");
405 
406  final Player p = PlayerUtils.getPlayer(player);
407  PreCon.isValid(p != null, "Invalid player");
408 
409  final Quest quest = getQuest(questPath, false);
410  assert quest != null;
411 
412  if (quest instanceof SubQuest) {
413 
414  SubQuest subQuest = (SubQuest)quest;
415  QuestStatus status = subQuest.getParent().getStatus(p);
416 
417  PreCon.isValid(status.getCurrentStatus() == CurrentQuestStatus.IN_PROGRESS,
418  "Cannot join a sub quest unless already joined to the parent quest.");
419  }
420 
421  UpdateSubscriber<IRequestContext> subscriber = new UpdateSubscriber<IRequestContext>() {
422  @Override
423  public void on(IRequestContext argument) {
424  quest.accept(p);
425  onAccept.run();
426  }
427  };
428 
429  REQUESTS.add(subscriber);
430 
431  ResponseRequestor.contextBuilder(ArborianQuests.getPlugin())
432  .name(questPath)
433  .timeout(15)
434  .response(ResponseType.ACCEPT)
435  .build(p)
436  .onRespond(subscriber)
437  .onRespond(new UpdateSubscriber<IRequestContext>() {
438  @Override

441  }
442  })
443  .onTimeout(new UpdateSubscriber<IRequestContext>() {
444  @Override
445  public void on(IRequestContext argument) {
446  Msg.tell(p, Lang.get(_REQUEST_EXPIRED));
447  }
448  })
