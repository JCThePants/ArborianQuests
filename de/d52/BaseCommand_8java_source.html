88  @Override
89  public void execute(CommandSender sender, ICommandArguments args) throws CommandException {
90 
91  CommandException.checkNotConsole(getPlugin(), this, sender);
92 
93  int page = args.getInteger("page");
94  final Player p = (Player)sender;
95 
96  ChatPaginator pagin = createPagin(args, 7, Lang.get(_PAGINATOR_TITLE));
97 
98  Collection<Quest> quests = ArborianQuests.getQuestManager().getQuests();
99  List<HierarchyNode<Quest>> nodes = new ArrayList<>(quests.size());
100 
101  for (Quest quest : quests) {
102 
103  // make sure the player is in the quest
104  QuestStatus status = quest.getStatus(p);
105  if (status.getCurrentStatus() == CurrentQuestStatus.IN_PROGRESS) {
106 
107  // check all sub nodes and remove quests the player is not in
108  HierarchyNode<Quest> node = new HierarchyNode<>(quest);
109  Iterator<TreeNode<Quest>> iterator = node.iterator();
110 
111  while (iterator.hasNext()) {
112 
113  TreeNode<Quest> questNode = iterator.next();
114  status = questNode.getValue().getStatus(p);
115 
116  Msg.debug(questNode.getValue().getDisplayName());
117 
118  if (status.getCurrentStatus() == CurrentQuestStatus.NONE) {
119  iterator.remove();
120  }
121  }
122 
123  nodes.add(node);
124  }
125  }
126 
127  // Create a chat hierarchy
128  ChatTree<Quest> questTree = new ChatTree<>(ArborianQuests.getPlugin());
129  questTree.addAllRootNodes(nodes);
130 
131  // add chat hierarchy to paginator
132  pagin.addAll(questTree.toChatLines(new NodeLineWriter<Quest>() {
133  @Override
134  public String write(Quest quest) {
135 
136  String line = quest instanceof PrimaryQuest
137  ? TextUtils.format(FormatTemplate.LIST_ITEM_DESCRIPTION,
138  quest.getDisplayName(), quest.getName()).toString()
139  : TextUtils.format("{WHITE}{0}", quest.getName()).toString();
140 
141  String objective = quest.getObjectives().getPlayerObjective(p.getUniqueId());
142 
143  if (objective != null) {
144  line += "{GRAY} - " + objective;
145  }
146 
147  return line;
148  }
149  }));
150 
151  pagin.show(p, page, FormatTemplate.RAW);
152 
153  tell(sender, Lang.get(_HELP, getInlineHelpCommand()));
154  }
155 }
